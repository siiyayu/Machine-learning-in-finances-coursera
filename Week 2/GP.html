//   ,  ,      data

    vector[N1] mu = rep_vector(.0, N1); // ,     
}


// , -

    real<lower=0> a;      // RBF 
    real<lower=0> l;      //  RBF 
    real<lower=0> sigma;  // 
    
}

//

    matrix[N1,N1] K; //    
    
    for (i in 2:N1)
        for(j in 1:(i-1))
        {
            K[i,j] = square(a) * exp(-square(x1[i]-x1[j])*inv_square(l*365)*0.5 );
            K[j,i] = K[i,j]; //   
        }
    
    for (n in 1:N1)
        K[n,n] = square(a) +
                 square(sigma) +   
                 1e-12;    //   ,         
}

//         . Stan -
//        . 
// ,

    y1 ~ multi_normal(rep_vector(0,N1),K); 
    
    
    a ~ normal(0, 1);  //    
    l ~ lognormal(2, .1); //   .

}

    vector[N1] mu = rep_vector(.0, N1); // ,     
}


// , -     
parameters {              // 
    real<lower=0> a;      // RBF 
    real<lower=0> l;      //  RBF 
    real<lower=0> sigma;  // 
    real<lower=0> b;      //  
    
}

//

    matrix[N1,N1] K; //    
    
    for (i in 2:N1)
        for(j in 1:(i-1))
        {
            K[i,j] = square(a) * exp(-square(x1[i]-x1[j])*inv_square(l*365)*0.5 ) +
